Classe Principal que mantém todos os clientes numa lista alfabética.

public class Agenda {
    // Consiste na utilização de um vetor de 27 elementos (total de letras do alfabéto + 1)
    // cada qual é uma lista para Cliente cujo nomes começam com a letra correspondente ao índice.
    private List<Client>[] agenda_ = new LinkedList[27];

    // O construtor default instância cada elemento do vetor como sendo do tipo LinkedList;
    public Agenda();

    // Busca de cliente por meio do nome.
    public Client findIt(String name);

    // Adiciona um novo cliente.
    public void add(Client cliente);

    // Remove o cliente forncendo o seu nome.
    public void remove(String name);

    // Retorna quantos clientes esão na lista
    public int Size();

    // Imprime todos os clientes.
    public void Print();

    // 
    // Foi identificado que para se obter o índice de uma letra no alfabéto,
    // basta subtrair 65 do valor da char, assim: 'A' == 0, 'B' == 1.
    // Exemplo:
    //		String name = "José";
    //		int index = getIndex(name);	// pega a primeira letra de name
    //		print(index);	// o resultado é 9
    public int getIndex(String str);
}


Exemplo de implementação:

	Agenda a = new Agenda();
	Date dat = new Date();
	Client cl1 = new Client("Tobias", dat);
	Client cl2 = new Client("Tania", dat);
	Client cl3 = new Client("José", dat, "Masculino");
	Client cl4 = new Client("Jana", dat, "Faminino");
	Client cl5 = new Client("Amanda", dat, "Feminino");

	a.add(cl1);
	a.add(cl2);
	a.add(cl3);
	a.add(cl4);
	a.add(cl5);

	a.Print();       // Imprime todos os clientes
        	         // em ordem alfabética
